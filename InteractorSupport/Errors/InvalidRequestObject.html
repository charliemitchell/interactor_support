def initialize(request_class:, errors: [])
  @request_class = request_class
  @errors = Array(errors)

  request_name =
    if request_class.respond_to?(:name) && request_class.name
      request_class.name
    else
      request_class.to_s
    end

  detail = @errors.any? ? ": #{@errors.join(', ')}" : ''

  super("Invalid #{request_name}#{detail}")
end
